## Benutzung

```tsx
import { Brand } from '@northware/ui/components/brand';

// ein festgelegter ServiceType
export default function Example() {
    return (
        <Brand service="cockpit"/>
    )
}

// ein geerberter ServiceType
export default function Example({service} : {service: ServiceType}) {
    return (
        <Brand service={service}/>
    )
}
```
