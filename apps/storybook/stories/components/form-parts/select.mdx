# Select

Zeigt eine Liste von Optionen, aus denen der User wählen kann, ausgeöst durch einen Button.

## Benutzung

```tsx
import {
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from "@northware/ui/components/select"

<Select>
  <SelectTrigger className="w-96">
    <SelectValue placeholder="Theme" />
  </SelectTrigger>
  <SelectContent>
    <SelectItem value="light">Light</SelectItem>
    <SelectItem value="dark">Dark</SelectItem>
    <SelectItem value="system">System</SelectItem>
  </SelectContent>
</Select>

```
